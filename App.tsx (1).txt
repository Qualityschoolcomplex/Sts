import { Switch, Route, Router as WouterRouter } from "wouter";
import { QueryClient, QueryClientProvider } from "@tanstack/react-query";
import { Toaster } from "@/components/ui/toaster";
import { TooltipProvider } from "@/components/ui/tooltip";
import { AuthProvider, useAuth } from "@/lib/auth";
import LoginPage from "@/pages/login";
import DashboardPage from "@/pages/dashboard";
import StudentsPage from "@/pages/students";
import StudentDetailPage from "@/pages/student-detail";
import ClassesPage from "@/pages/classes";
import TermsPage from "@/pages/terms";
import SubjectsPage from "@/pages/subjects";
import ResultsPage from "@/pages/results";
import ReportCardPage from "@/pages/report-card";
import UsersManagementPage from "@/pages/users-management";
import Layout from "@/components/layout";
import NotFound from "@/pages/not-found";

const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      retry: 1,
      staleTime: 30000,
    },
  },
});

function AuthenticatedRoutes() {
  const { user, isLoading } = useAuth();

  if (isLoading) {
    return (
      <div className="min-h-screen flex items-center justify-center bg-background">
        <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-primary" />
      </div>
    );
  }

  if (!user) {
    return <LoginPage />;
  }

  return (
    <Layout>
      <Switch>
        <Route path="/" component={DashboardPage} />
        <Route path="/students" component={StudentsPage} />
        <Route path="/students/:id">{(params) => <StudentDetailPage id={params.id} />}</Route>
        <Route path="/classes" component={ClassesPage} />
        <Route path="/terms" component={TermsPage} />
        <Route path="/subjects" component={SubjectsPage} />
        <Route path="/results" component={ResultsPage} />
        <Route path="/reports/:studentId/:termId">
          {(params) => <ReportCardPage studentId={params.studentId} termId={params.termId} />}
        </Route>
        <Route path="/users" component={UsersManagementPage} />
        <Route component={NotFound} />
      </Switch>
    </Layout>
  );
}

function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <TooltipProvider>
        <WouterRouter base={import.meta.env.BASE_URL.replace(/\/$/, "")}>
          <AuthProvider>
            <AuthenticatedRoutes />
          </AuthProvider>
        </WouterRouter>
        <Toaster />
      </TooltipProvider>
    </QueryClientProvider>
  );
}

export default App;
